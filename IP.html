网络层位于OSI七层模型的第三层，其特点是不可靠、无连接，重要的功能是路由分发。其数据格式如下：

IP报文格式

现在，请按该格式补充代码文件net.h的ip结构体， 以及net_tx_ip方法发送数据时补充相应字段信息、net_rx_ip接收数据时可能也需要补充一些字段判断的相关逻辑。注意到，大部分的PC端系统都是小端架构，而网络传输统一是大端，即高低位字节的分布不一致，所以在发送和接收到数据时，需要调用htonl（32位）或htons（16位）转换数据，样例见源IP地址字段。

// an IP packet header (comes after an Ethernet header).
struct ip {
  // todo: 补充源IP地址前面的剩余字段
  uint32 ip_src, ip_dst;
};

// sends an IP packet
static void
net_tx_ip(struct mbuf *m, uint8 proto, uint32 dip)
{
  struct ip *iphdr;

  // push the IP header
  iphdr = mbufpushhdr(m, *iphdr);
  memset(iphdr, 0, sizeof(*iphdr));

  // todo：补充相应字段

  iphdr->ip_src = htonl(local_ip);
  iphdr->ip_dst = htonl(dip);

  // now on to the ethernet layer
  net_tx_eth(m, ETHTYPE_IP);
}

// receives an IP packet
static void
net_rx_ip(struct mbuf *m)
{
  struct ip *iphdr;
  uint16 len;

  iphdr = mbufpullhdr(m, *iphdr);
  if (!iphdr)
      goto fail;

  // todo: 可能需要补充一些字段逻辑，但是作为实验demo，暂时忽略也能正常运行

  // is the packet addressed to us?
  if (htonl(iphdr->ip_dst) != local_ip)
    goto fail;

  len = ntohs(iphdr->ip_len) - sizeof(*iphdr);
  net_rx_udp(m, len, iphdr);
  return;

fail:
  mbuffree(m);
}
